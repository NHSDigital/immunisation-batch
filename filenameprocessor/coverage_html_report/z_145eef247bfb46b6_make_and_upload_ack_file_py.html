33def upload_ack_file(file_key: str, ack_data: dict) -> None: 

34 """Formats the ack data into a csv file and uploads it to the ack bucket""" 

35 ack_filename = "ack/" + file_key.replace(".csv", "_InfAck.csv") 

36 

37 # Create CSV file with | delimiter, filetype .csv 

38 csv_buffer = StringIO() 

39 csv_writer = writer(csv_buffer, delimiter="|") 

40 csv_writer.writerow(list(ack_data.keys())) 

41 csv_writer.writerow(list(ack_data.values())) 

42 

43 # Upload the CSV file to S3 

44 csv_buffer.seek(0) 

45 csv_bytes = BytesIO(csv_buffer.getvalue().encode("utf-8")) 

46 ack_bucket_name = os.getenv("ACK_BUCKET_NAME", f"immunisation-batch-{get_environment()}-data-destinations") 

47 s3_client.upload_fileobj(csv_bytes, ack_bucket_name, ack_filename)
